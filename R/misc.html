1 Output an R object as code

You can use dput to return the object to the console, or dump to write it to a file. 

X <- data.frame(X1 = 1:5, X2 = c("A", "B", "C", "D", "E"), X3 = rnorm(5))
dput(X)

structure(list(X1 = 1:5, X2 = structure(1:5, .Label = c("A", 
"B", "C", "D", "E"), class = "factor"), X3 = c(-0.391688657925633, 
-1.5912293387651, 0.0731426650249493, -0.571196724364881, -1.635197314415
)), class = "data.frame", row.names = c(NA, -5L))

4 Check whether a matrix is symmetric along the diagonal

isSymmetric <- function(X) {
    all(X == t(X))
}

I <- diag(3)
isSymmetric(I)

I[1,2] <- 1
isSymmetric(I)

[1] TRUE
[1] FALSE

10 Calculate change scores in dplyr

library(dplyr)

set.seed(1)

study <- rep(1:2, each=4)
drug <- rep(1:2, 2, each=2)
timepoint <- rep(1:2, 4)
mean <- round(rnorm(8), 2)
data <- data.frame(study, drug, timepoint, mean)

data %>%
    group_by(study, drug) %>%
    summarise(change = mean[timepoint == 2] - mean[timepoint == 1])
study drug change
1 1 0.81
1 2 2.44
2 1 -1.15
2 2 0.25

13 Collect non-NA values from two or more vectors

This function is called coalesce in SQL: 

coalesce <- function(...) {
  Reduce(function(x, y) ifelse(is.na(x), y, x),
	 list(...))
}

x1 <- c(0, NA, NA)
x2 <- c(NA, 1, NA)
x3 <- c(NA, NA, 2)

coalesce(x1, x2, x3)

[1] 0 1 2
